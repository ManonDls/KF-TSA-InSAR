[docs]    def run_lq(self, m0, P0, add_err, gamma=1.):
        ''' 
        Linear regression with least square accounting for covariances in data and model 
        (after building them).
            * m0 : initial/prior values for parameters (1D array or list)
            * P0 : initial/prior covariance matrix for model parameters (2D square array)
            * add_err : last element of diagonal in process noise covariance
            * gamma : the weight of the deformation with respect to the phase retrieval'''
        
        '''
        #Check for NaN in interferograms
        mask_nan  = np.isnan(self.D) #True if NaN
        ind_nan   = np.array(range(len(self.D)))[mask_nan]
        if len(ind_nan) > 0 : 
            self.D = np.delete(self.D,ind_nan)
            self.H = np.delete(self.H,ind_nan,0) 
            self.R = np.delete(self.R,ind_nan,0)
            self.R = np.delete(self.R,ind_nan,1) 
        '''
        
        #Build initial model such that d = Am 
        #(interf_21 = phase2-phase1 and phase1 = m1*x+ m2*sin(x)+m3*cos(x)...) 
        mp = np.concatenate(( np.zeros(len(self.t)),m0 ))
        
        if self.D.ndim >2:
            ny,nx = np.shape(self.D)[:2]
            self.D = np.reshape(self.D,(ny*nx,self.H.shape[0])) #flatten 2 first dimensions
        
        if self.D.ndim >1:
            y = np.concatenate((self.D,np.zeros((np.shape(self.D)[0],len(self.t)))),axis=1)     
        else :
            y = np.concatenate(( self.D,np.zeros(len(self.t)) ))  
            
        ### Matrix of model A in 4 quarter : ((A1,A2),(A3,A4))
        A1 = self.H[:,:]                              #(n_data x n_times)
        A2 = np.zeros((np.shape(self.D)[-1],len(m0))) #(n_data x n_params)
        A3 = np.eye(len(self.t))*(-1.)                 #(n_times x n_times)
        A4 = np.zeros((len(self.t),len(m0)))          #(n_times x n_params) 
        
        A4 = self.transition_vect(self.t).T
        A = np.vstack(( np.hstack((A1,A2)), gamma*np.hstack((A3,A4)) ))
        
        #TEST: add first row to set phi_0 to zero
        row0 = np.zeros(len(self.t)+len(m0))
        row0[0] = 1.
        A = np.vstack(( row0,A ))
        y = np.concatenate((np.array([np.zeros(np.shape(y[...,0]))]).T,y),axis=-1)
        
        #######
        self.A = A
        
        ### Weight matrix for covariance in Data
        Cd_inv = self.build_Cd_inv(add_err)
             
        ### Regularisation / covariance in model 
        Cmp_inv = self.build_Cmp_inv(P0,len(mp))
        
        ### Least square regression 
        Cm = np.linalg.inv(np.dot(np.dot(A.T,Cd_inv),A) +Cmp_inv) #posterior model covariance
        
        if self.D.ndim ==2:
            ATb  = [np.dot(np.dot(A.T, Cd_inv), y[i,:]) for i in range(np.shape(self.D)[0])]
            m    =  np.array([np.dot(Cm, ATb[i]) for i in range(np.shape(self.D)[0])]) 
            self.state = np.concatenate((m[:,len(self.t):],m[:,:len(self.t)]),axis=1)
        else :
            ATb  = np.dot(np.dot(A.T, Cd_inv), (y-np.dot(A,mp))) 
            m    = mp + np.dot(Cm, ATb)                 #adjustment
            self.m = m 
            self.state = np.concatenate((m[len(self.t):],m[:len(self.t)]))
        
        self.err = np.concatenate((np.sqrt(np.diag(Cm)[len(self.t):]),np.sqrt(np.diag(Cm)[:len(self.t)])))
        self.Cm = Cm
        
        return self.state, self.err

        
   

[docs]    def build_Cd_inv(self,err):
        ''' 
        Build composite inverse of data covariance (C_d)**(-1) from 'err' and 'R'
        Has dimention of the number of interfero + number of phases
            * err : the error associated with the lastly added phase (float) in KF '''
        
        N = np.shape(self.D)[-1]+1
        W = np.eye(N +len(self.t))

        if np.linalg.cond(self.R) < old_div(1,sys.float_info.epsilon) :    #test if singular R 
            W[1:N,1:N] = np.linalg.pinv((self.R))  #on data (interfero) 
            W[0,0] = old_div(1000.,(self.R[0,0]))            #first phase with 100 times smaller uncertainty than other phases
        else :
            print('R is singular')
            
        if err != 0 : 
            W[N:,N:] = W[N:,N:]*(err)**(-1) #on parameters 
        
       
        return W

        
        

[docs]    def build_Cmp_inv(self,P0,leng):
        '''
        Build the inverse of the prior model covariance matrix (C_m_p)**(-1)
            * P0 : prior covariance of model parameters (2D array) (LxL)
            * leng : length of the state vector m (int) (len(t)+L)'''
        
        Wr = np.zeros((leng,leng))
        if P0.any() :                          #False if all zero
            Wr[len(self.t):,len(self.t):] = np.linalg.inv(P0) 
        
        return Wr


         

[docs]    def test_misfit(self,d,Cd,m,Cm_p,A):
        ''' 
        Verify the fact that (Tarantola 3.3.4): 
        The minimum of twice the misfit function is distributed following 
        a chi2 distribution with len(d) degrees of freedom '''
        
        #Twice the misfit function
        msft2 = np.dot(np.dot((np.dot(A,m)-d).T, np.linalg.inv(
                    np.dot(np.dot(A, Cm_p), A.T) + Cd)), d)
        msft2 = abs(msft2)
        
        freed_deg = len(d) - np.linalg.matrix_rank(A) #p177 Tarantola
        Chi_ref = max(chi2.pdf(m,freed_deg))
        
        if (msft2 < 1/2.*Chi_ref) or (msft2 > 2*Chi_ref):
            print("WARNING : May be a problem in least square approximation")  
            print("          The misfit is",msft2,"while the X2 is",Chi_ref)

        ''' 
        Returns the array of reference parameters with sin and cos amplitudes 
        transformed into amplitude and phase shift of oscillation AND 
        the phases according to the model
        * i,j : indxs when 2D to select pixel, not used otherwise (x,y)'''
        
        
        if self.state.ndim==1 :
            m_pshift,tmp = self.comp_phase_shift(self.state[:self.L])
            model = self.draw_model(self.state[:self.L])
        if self.state.ndim==2 :
            self.state = np.reshape(self.state,(Ny,Nx,len(self.t)+self.L))   
        if self.state.ndim==3 :
            m_pshift,tmp = self.comp_phase_shift(self.state[j,i,:self.L])
            model = self.draw_model(self.state[j,i,:self.L])

        return m_pshift, model

        '''Compute condition number of problem'''
        
        cn = np.linalg.norm(self.A)*np.linalg.norm(np.linalg.pinv(self.A))
        #or np.linalg.cond(self.A)
